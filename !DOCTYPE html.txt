    <script>
        // Data structures for deadlock detection
        let processes = new Set();
        let resources = new Map();
        let allocation = new Map();  // Map of process -> Map of resource -> count
        let request = new Map();     // Map of process -> Map of resource -> count
        let available = new Map();   // Map of resource -> available count
        let totalResources = new Map(); // Map of resource -> total count
        
        // Initialize resources
        function initializeResources() {
            const resourceTypes = ['R0', 'R1', 'R2', 'R3'];
            resourceTypes.forEach(res => {
                totalResources.set(res, 5);
                available.set(res, 5);
                resources.set(res, { total: 5, available: 5 });
            });
        }
        
        // Add a new process
        function addProcess() {
            const processId = document.getElementById('processId').value.trim();
            if (!processId) {
                alert('Please enter a process ID');
                return;
            }
            
            if (processes.has(processId)) {
                alert('Process already exists!');
                return;
            }
            
            processes.add(processId);
            allocation.set(processId, new Map());
            request.set(processId, new Map());
            
            // Initialize allocation and request for this process
            for (let res of totalResources.keys()) {
                allocation.get(processId).set(res, 0);
                request.get(processId).set(res, 0);
            }
            
            updateDisplay();
            detectDeadlock();
        }
        
        // Remove a process
        function removeProcess() {
            const processId = document.getElementById('processId').value.trim();
            if (!processId || !processes.has(processId)) {
                alert('Process not found!');
                return;
            }
            
            // Release all resources held by this process
            const allocMap = allocation.get(processId);
            for (let [res, count] of allocMap) {
                if (count > 0) {
                    available.set(res, available.get(res) + count);
                    resources.get(res).available += count;
                }
            }
            
            processes.delete(processId);
            allocation.delete(processId);
            request.delete(processId);
            
            updateDisplay();
            detectDeadlock();
        }
        
        // Perform request or release operation
        function performOperation() {
            const processId = document.getElementById('processId').value.trim();
            const resource = document.getElementById('resourceType').value;
            const operation = document.getElementById('operation').value;
            
            if (!processId || !processes.has(processId)) {
                alert('Please add the process first!');
                return;
            }
            
            if (operation === 'request') {
                requestResource(processId, resource);
            } else {
                releaseResource(processId, resource);
            }
        }
        
        // Request resource with Banker's Algorithm check
        function requestResource(processId, resource) {
            const currentRequest = request.get(processId).get(resource) + 1;
            const currentAlloc = allocation.get(processId).get(resource);
            
            if (currentRequest > totalResources.get(resource)) {
                alert(`Cannot request more than total available (${totalResources.get(resource)})`);
                return;
            }
            
            // Check if request can be granted
            if (available.get(resource) > 0 && currentRequest <= available.get(resource)) {
                // Temporarily allocate for safety check
                const oldAvailable = available.get(resource);
                available.set(resource, available.get(resource) - 1);
                request.get(processId).set(resource, currentRequest);
                allocation.get(processId).set(resource, currentAlloc + 1);
                
                // Check if system is in safe state
                if (isSafeState()) {
                    resources.get(resource).available = available.get(resource);
                    updateDisplay();
                    showMessage(`✓ Request granted: ${processId} requested ${resource}`, 'success');
                } else {
                    // Rollback if unsafe
                    available.set(resource, oldAvailable);
                    request.get(processId).set(resource, currentRequest - 1);
                    allocation.get(processId).set(resource, currentAlloc);
                    showMessage(`⚠ Request denied: Would lead to unsafe state!`, 'error');
                }
            } else {
                showMessage(`❌ Request denied: Insufficient resources available`, 'error');
            }
            
            detectDeadlock();
        }
        
        // Release resource
        function releaseResource(processId, resource) {
            const currentAlloc = allocation.get(processId).get(resource);
            
            if (currentAlloc > 0) {
                allocation.get(processId).set(resource, currentAlloc - 1);
                request.get(processId).set(resource, Math.max(0, request.get(processId).get(resource) - 1));
                available.set(resource, available.get(resource) + 1);
                resources.get(resource).available = available.get(resource);
                updateDisplay();
                showMessage(`✓ Released: ${processId} released ${resource}`, 'success');
            } else {
                showMessage(`No resources to release for ${processId}`, 'error');
            }
            
            detectDeadlock();
        }
        
        // Banker's Algorithm for safe state detection
        function isSafeState() {
            let work = new Map(available);
            let finish = new Map();
            let safeSequence = [];
            
            for (let p of processes) {
                finish.set(p, false);
            }
            
            let found;
            do {
                found = false;
                for (let p of processes) {
                    if (!finish.get(p)) {
                        let canAllocate = true;
                        let need = request.get(p);
                        
                        for (let [res, needCount] of need) {
                            if (needCount > (work.get(res) || 0)) {
                                canAllocate = false;
                                break;
                            }
                        }
                        
                        if (canAllocate) {
                            for (let [res, allocCount] of allocation.get(p)) {
                                work.set(res, work.get(res) + allocCount);
                            }
                            finish.set(p, true);
                            safeSequence.push(p);
                            found = true;
                        }
                    }
                }
            } while (found);
            
            // Check if all processes can finish
            let allFinished = true;
            for (let p of processes) {
                if (!finish.get(p)) {
                    allFinished = false;
                    break;
                }
            }
            
            if (allFinished) {
                document.getElementById('safeSequenceResult').innerHTML = 
                    `<div class="safe-sequence">
                        <strong>✅ Safe Sequence Found:</strong> ${safeSequence.join(' → ')}
                    </div>`;
                return true;
            } else {
                document.getElementById('safeSequenceResult').innerHTML = 
                    `<div class="deadlock-alert">
                        <strong>⚠ No Safe Sequence Found!</strong> System may be in unsafe state.
                    </div>`;
                return false;
            }
        }
        
        // Deadlock detection using Wait-For Graph
        function detectDeadlock() {
            // Build wait-for graph
            let waitFor = new Map();
            for (let p of processes) {
                waitFor.set(p, new Set());
            }
            
            // Check for circular wait
            for (let p of processes) {
                for (let [res, needCount] of request.get(p)) {
                    if (needCount > 0) {
                        // Find which process holds this resource
                        for (let q of processes) {
                            if (q !== p && allocation.get(q).get(res) > 0) {
                                waitFor.get(p).add(q);
                            }
                        }
                    }
                }
            }
            
            // Detect cycles using DFS
            let visited = new Set();
            let recStack = new Set();
            let deadlocked = false;
            
            function hasCycle(node) {
                if (!visited.has(node)) {
                    visited.add(node);
                    recStack.add(node);
                    
                    for (let neighbor of waitFor.get(node)) {
                        if (!visited.has(neighbor) && hasCycle(neighbor)) {
                            return true;
                        } else if (recStack.has(neighbor)) {
                            return true;
                        }
                    }
                }
                recStack.delete(node);
                return false;
            }
            
            for (let p of processes) {
                if (hasCycle(p)) {
                    deadlocked = true;
                    break;
                }
            }
            
            const statusElement = document.getElementById('deadlockStatus');
            const resultDiv = document.getElementById('detectionResult');
            
            if (deadlocked) {
                statusElement.innerHTML = '⚠ Deadlock!';
                statusElement.style.color = '#ff4444';
                resultDiv.innerHTML = `
                    <div class="deadlock-alert">
                        <strong>🚨 DEADLOCK DETECTED!</strong><br>
                        Circular wait condition found. Recovery strategies:<br>
                        • Terminate one or more processes in the cycle<br>
                        • Preempt resources from deadlocked processes<br>
                        • Rollback processes to previous safe state
                    </div>`;
            } else {
                statusElement.innerHTML = '✓ Safe';
                statusElement.style.color = '#28a745';
                resultDiv.innerHTML = `
                    <div class="safe-sequence">
                        <strong>✅ No Deadlock Detected</strong><br>
                        System is in a safe state.
                    </div>`;
            }
            
            drawResourceGraph(waitFor);
        }
        
        // Find and display safe sequence
        function findSafeSequence() {
            isSafeState();
        }
        
        // Draw resource allocation graph
        function drawResourceGraph(waitFor) {
            const canvas = document.getElementById('resourceGraph');
            if (!canvas) return;
            
            const ctx = canvas.getContext('2d');
            canvas.width = canvas.parentElement.clientWidth;
            canvas.height = 400;
            
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            if (processes.size === 0) {
                ctx.fillStyle = '#999';
                ctx.font = '16px Arial';
                ctx.fillText('No processes to display', canvas.width/2 - 100, canvas.height/2);
                return;
            }
            
            // Draw processes as circles
            const processPositions = new Map();
            const radius = 30;
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;
            const angleStep = (2 * Math.PI) / processes.size;
            let angle = 0;
            
            // Position processes in a circle
            let index = 0;
            for (let p of processes) {
                const x = centerX + 150 * Math.cos(angle);
                const y = centerY + 150 * Math.sin(angle);
                processPositions.set(p, { x, y });
                
                // Draw process node
                ctx.beginPath();
                ctx.arc(x, y, radius, 0, 2 * Math.PI);
                ctx.fillStyle = '#667eea';
                ctx.fill();
                ctx.strokeStyle = '#333';
                ctx.stroke();
                ctx.fillStyle = 'white';
                ctx.font = 'bold 16px Arial';
                ctx.fillText(p, x - 10, y + 5);
                
                angle += angleStep;
                index++;
            }
            
            // Draw edges (wait-for relationships)
            for (let [from, toSet] of waitFor) {
                const fromPos = processPositions.get(from);
                if (!fromPos) continue;
                
                for (let to of toSet) {
                    const toPos = processPositions.get(to);
                    if (!toPos) continue;
                    
                    ctx.beginPath();
                    ctx.moveTo(fromPos.x, fromPos.y);
                    ctx.lineTo(toPos.x, toPos.y);
                    ctx.strokeStyle = '#ff4444';
                    ctx.lineWidth = 2;
                    ctx.stroke();
                    
                    // Draw arrow
                    const angle2 = Math.atan2(toPos.y - fromPos.y, toPos.x - fromPos.x);
                    const arrowX = toPos.x - radius * Math.cos(angle2);
                    const arrowY = toPos.y - radius * Math.sin(angle2);
                    ctx.beginPath();
                    ctx.moveTo(arrowX, arrowY);
                    ctx.lineTo(arrowX - 8 * Math.cos(angle2 - Math.PI/6), arrowY - 8 * Math.sin(angle2 - Math.PI/6));
                    ctx.lineTo(arrowX - 8 * Math.cos(angle2 + Math.PI/6), arrowY - 8 * Math.sin(angle2 + Math.PI/6));
                    ctx.fillStyle = '#ff4444';
                    ctx.fill();
                }
            }
        }
        
        // Update all displays
        function updateDisplay() {
            // Update process list
            const processesDiv = document.getElementById('processes');
            processesDiv.innerHTML = '';
            for (let p of processes) {
                processesDiv.innerHTML += `<div class="process-item">
                    <h4>${p}</h4>
                    <small>Allocated: ${Array.from(allocation.get(p).entries()).map(([r,c]) => `${r}:${c}`).join(', ')}</small><br>
                    <small>Requesting: ${Array.from(request.get(p).entries()).map(([r,c]) => `${r}:${c}`).join(', ')}</small>
                </div>`;
            }
            
            // Update resource list
            const resourcesDiv = document.getElementById('resources');
            resourcesDiv.innerHTML = '';
            for (let [res, data] of resources) {
                resourcesDiv.innerHTML += `<div class="resource-item">
                    <h4>${res}</h4>
                    <div>Available: ${data.available} / ${data.total}</div>
                </div>`;
            }
            
            // Update statistics
            document.getElementById('totalProcesses').textContent = processes.size;
            document.getElementById('totalResources').textContent = resources.size;
            
            // Update allocation matrix
            if (processes.size > 0) {
                let allocHtml = '<table class="allocation-table"><tr><th>Process</th>';
                for (let res of totalResources.keys()) {
                    allocHtml += `<th>${res}</th>`;
                }
                allocHtml += '</tr>';
                
                for (let p of processes) {
                    allocHtml += `<tr><td><strong>${p}</strong></td>`;
                    for (let res of totalResources.keys()) {
                        allocHtml += `<td>${allocation.get(p).get(res) || 0}</td>`;
                    }
                    allocHtml += '</tr>';
                }
                allocHtml += '</table>';
                document.getElementById('allocationMatrix').innerHTML = allocHtml;
                
                // Update request matrix
                let reqHtml = '<table class="allocation-table"><tr><th>Process</th>';
                for (let res of totalResources.keys()) {
                    reqHtml += `<th>${res}</th>`;
                }
                reqHtml += '</tr>';
                
                for (let p of processes) {
                    reqHtml += `<tr><td><strong>${p}</strong></td>`;
                    for (let res of totalResources.keys()) {
                        reqHtml += `<td>${request.get(p).get(res) || 0}</td>`;
                    }
                    reqHtml += '</tr>';
                }
                reqHtml += '</table>';
                document.getElementById('requestMatrix').innerHTML = reqHtml;
            } else {
                document.getElementById('allocationMatrix').innerHTML = '<p>No processes added</p>';
                document.getElementById('requestMatrix').innerHTML = '<p>No processes added</p>';
            }
        }
        
        // Show message
        function showMessage(msg, type) {
            const resultDiv = document.getElementById('detectionResult');
            const className = type === 'success' ? 'safe-sequence' : 'deadlock-alert';
            resultDiv.innerHTML = `<div class="${className}">${msg}</div>`;
            setTimeout(() => {
                if (document.getElementById('detectionResult').innerHTML === resultDiv.innerHTML) {
                    // Keep the message, don't clear automatically
                }
            }, 3000);
        }
        
        // Reset simulation
        function resetSimulation() {
            processes.clear();
            allocation.clear();
            request.clear();
            initializeResources();
            updateDisplay();
            detectDeadlock();
            showMessage('Simulation reset successfully', 'success');
        }
        
        // Initialize the simulator
        initializeResources();
        updateDisplay();

            addProcess();
        }, 100);
